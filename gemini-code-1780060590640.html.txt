    <script>
        let score = 0;
        let timeLeft = 10;

        let correctAnswer = 0;
        let username = "";
        let playerToken = ""; 
        
        let isBGMOn = localStorage.getItem('bgm_on') !== 'false';
        let isSFXOn = localStorage.getItem('sfx_on') !== 'false';
        let audioCtx = null;
        let bgmInterval = null;

        let deleteTimer;
        let deleteSecondsLeft = 5;

        // คลาวด์กลางในการซิงค์ข้อมูลร่วมกันระหว่างมือถือทุกยี่ห้อ
        const API_URL = "https://kvbin.org/v1/mathgame_premium_board";

        // ระบบจับรหัสประจำเครื่องที่เป็นเอกลักษณ์ (ใช้งานได้มือถือทุกยี่ห้อในโลก)
        function initPlayerToken() {
            playerToken = localStorage.getItem('player_secret_token');
            if (!playerToken) {
                playerToken = 'p_' + Math.random().toString(36).substring(2, 15) + Date.now().toString(36);
                localStorage.setItem('player_secret_token', playerToken);
            }
        }

        function initAudio() {
            if (!audioCtx) audioCtx = new (window.AudioContext || window.webkitAudioContext)();
            if (audioCtx && audioCtx.state === 'suspended') audioCtx.resume();
        }

        function playTone(freq, type, duration) {
            if (!isSFXOn) return;
            initAudio();
            try {
                if (!audioCtx) return;
                let osc = audioCtx.createOscillator();
                let gain = audioCtx.createGain();
                osc.type = type;
                osc.frequency.value = freq;
                gain.gain.setValueAtTime(0.25, audioCtx.currentTime); 
                gain.gain.exponentialRampToValueAtTime(0.00001, audioCtx.currentTime + duration);
                osc.connect(gain);
                gain.connect(audioCtx.destination);
                osc.start();
                osc.stop(audioCtx.currentTime + duration);
            } catch(e) {}
        }

        function playCorrectSound() { playTone(987.77, 'sine', 0.12); setTimeout(() => playTone(1318.51, 'sine', 0.18), 80); }
        function playWrongSound() { playTone(150, 'sawtooth', 0.4); }

        function playBGMTrack() {
            if (!isBGMOn) return;
            initAudio();
            if (!audioCtx) return;
            const notes = [261.63, 329.63, 392.00, 523.25, 392.00, 329.63, 440.00, 349.23, 392.00, 293.66];
            let now = audioCtx.currentTime;
            notes.forEach((note, index) => {
                try {
                    let osc = audioCtx.createOscillator();
                    let gain = audioCtx.createGain();
                    osc.type = 'triangle';
                    osc.frequency.value = note;
                    gain.gain.setValueAtTime(0.06, now + (index * 0.35)); 
                    gain.gain.exponentialRampToValueAtTime(0.00001, now + (index * 0.35) + 0.3);
                    osc.connect(gain);
                    gain.connect(audioCtx.destination);
                    osc.start(now + (index * 0.35));
                    osc.stop(now + (index * 0.35) + 0.3);
                } catch(e){}
            });
        }

        function startBGM() {
            if (bgmInterval) clearInterval(bgmInterval);
            if (!isBGMOn) return;
            playBGMTrack();
            bgmInterval = setInterval(playBGMTrack, 3200);
        }

        function updateAudioButtons() {
            document.getElementById('bgm-toggle').innerText = isBGMOn ? '🎵' : '❌🎵';
            document.getElementById('sfx-toggle').innerText = isSFXOn ? '🔊' : '❌🔊';
        }

        function toggleBGM() {
            isBGMOn = !isBGMOn;
            localStorage.setItem('bgm_on', isBGMOn);
            updateAudioButtons();
            if (isBGMOn) startBGM(); else clearInterval(bgmInterval);
        }

        function toggleSFX() {
            isSFXOn = !isSFXOn;
            localStorage.setItem('sfx_on', isSFXOn);
            updateAudioButtons();
        }

        const emojis = ['🎈', '🌈', '🔥', '✨', '🍭', '🍕', '🐱', '🦄', '⭐', '🍀', '🍓', '🍩', '🧠', '⚡'];
        function createFloatingEmoji() {
            const container = document.getElementById('emoji-container');
            if (!container) return;
            const emoji = document.createElement('div');
            emoji.className = 'emoji-bg';
            emoji.innerText = emojis[Math.floor(Math.random() * emojis.length)];

            emoji.style.fontSize = size + 'px';
            emoji.style.animationDuration = duration + 's';
            container.appendChild(emoji);
            setTimeout(() => { emoji.remove(); }, duration * 1000);
        }

        function showScreen(screenId) {
            document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));

        }

        function startGame() {
            username = document.getElementById('username').value.trim();
            if (!username) {
                alert("กรุณาใส่ชื่อเล่นก่อนนะค้าบ! 🥰");
                return;
            }
            localStorage.setItem('last_active_username', username);
            initAudio();
            if (isBGMOn && !bgmInterval) startBGM();

            score = 0;
            document.getElementById('current-score').innerText = score;
            document.getElementById('my-highscore').innerText = localStorage.getItem('personal_high_token_score') || 0;

            showScreen('game-screen');
            nextQuestion();
        }

        function startQuestionTimer() {
            clearInterval(timerInterval);
            timeLeft = 10;
            updateTimerBar();
            timerInterval = setInterval(() => {
                timeLeft -= 0.1;
                updateTimerBar();
                if (timeLeft <= 0) {
                    clearInterval(timerInterval);
                    playWrongSound();
                    endGame(); 
                }
            }, 100);
        }

        function updateTimerBar() {
            document.getElementById('timer-bar').style.width = Math.max(0, (timeLeft / 10) * 100) + '%';
        }

        function generateQuestion() {
            let type = Math.floor(Math.random() * 5);
            let displayText = "";

            if (type === 0) { 
                let n1 = Math.floor(Math.random() * 40) + 5;
                let n2 = Math.floor(Math.random() * 40) + 5;
                correctAnswer = n1 + n2;
                displayText = `${n1} + ${n2}`;
            } else if (type === 1) { 
                let n1 = Math.floor(Math.random() * 50) + 10;
                let n2 = Math.floor(Math.random() * n1); 
                correctAnswer = n1 - n2;
                displayText = `${n1} - ${n2}`;
            } else if (type === 2) { 
                let n1 = Math.floor(Math.random() * 12) + 2;
                let n2 = Math.floor(Math.random() * 10) + 2;
                correctAnswer = n1 * n2;
                displayText = `${n1} × ${n2}`;
            } else if (type === 3) { 
                correctAnswer = Math.floor(Math.random() * 10) + 2;
                let n2 = Math.floor(Math.random() * 10) + 2;
                let n1 = correctAnswer * n2;
                displayText = `${n1} ÷ ${n2}`;
            } else { 
                if (Math.floor(Math.random() * 2) === 0) { 
                    let a = Math.floor(Math.random() * 6) + 1;
                    let b = Math.floor(Math.random() * 6) + 1;
                    let c = Math.floor(Math.random() * 4) + 2;
                    correctAnswer = (a + b) * c;
                    displayText = `(${a} + ${b}) × ${c}`;
                } else { 
                    let a = Math.floor(Math.random() * 15) + 5;
                    let b = Math.floor(Math.random() * a);
                    let c = Math.floor(Math.random() * 20) + 2;
                    correctAnswer = (a - b) + c;
                    displayText = `(${a} - ${b}) + ${c}`;
                }
            }

            document.getElementById('question').innerText = displayText;
            let choices = [correctAnswer];
            while (choices.length < 4) {
                let offset = (Math.floor(Math.random() * 10) + 1) * (Math.random() > 0.5 ? 1 : -1);
                let wrongAnswer = correctAnswer + offset;
                if (wrongAnswer >= 0 && !choices.includes(wrongAnswer)) choices.push(wrongAnswer);
            }

            choices.sort(() => Math.random() - 0.5);
            document.querySelectorAll('.option-btn').forEach((btn, idx) => {
                btn.innerText = choices[idx];
                btn.blur();
            });
        }

        function nextQuestion() { generateQuestion(); startQuestionTimer(); }

        function checkAnswer(button) {
            clearInterval(timerInterval);
            if (parseInt(button.innerText) === correctAnswer) {
                score++;
                document.getElementById('current-score').innerText = score;
                playCorrectSound();
                nextQuestion(); 
            } else {
                playWrongSound();
                endGame(); 
            }
        }

        function endGame() {
            clearInterval(timerInterval);
            showScreen('end-screen');
            document.getElementById('final-score').innerText = `คุณได้คะแนน: ${score} คะแนน`;
            
            let personalHigh = parseInt(localStorage.getItem('personal_high_token_score')) || 0;
            if (score > personalHigh) localStorage.setItem('personal_high_token_score', score);

            let medal = "🥉", rank = "ระดับฝึกหัด 🐣";
            if (score >= 20) { medal = "👑"; rank = "มหาเทพคณิตศาสตร์ 🪐"; }
            else if (score >= 12) { medal = "🥇"; rank = "อัจฉริยะคิดไว ⚡"; }
            else if (score >= 5) { medal = "🥈"; rank = "นักคิดเลขเร็ว 🏃‍♂️"; }
            
            document.getElementById('medal-icon').innerText = medal;
            document.getElementById('rank-title').innerText = `ฉายา: ${rank}`;
            saveScoreToCloud(username.trim(), score);
        }

        function renderLeaderboardTable(data) {
            if (!data || !Array.isArray(data)) return;
            let validData = data.filter(item => item && item.name).sort((a, b) => (parseInt(b.score) || 0) - (parseInt(a.score) || 0));
            let html = '';
            const topPlayers = validData.slice(0, 15); 
            
            if (topPlayers.length === 0) {
                html = `<tr><td colspan="3">ยังไม่มีข้อมูลอันดับ (ตารางว่างเปล่า)</td></tr>`;
            } else {
                topPlayers.forEach((player, index) => {
                    let rankEmoji = index === 0 ? '🥇' : index === 1 ? '🥈' : index === 2 ? '🥉' : index + 1;
                    let rowClass = index < 3 ? 'class="top-three"' : '';
                    if (player.token === playerToken) rowClass = 'style="background-color: #ffeff2; font-weight: bold; color: #ff4a75;"';
                    html += `<tr ${rowClass}><td>${rankEmoji}</td><td>${escapeHTML(player.name)}</td><td>${player.score}</td></tr>`;
                });
            }
            document.getElementById('start-leaderboard-body').innerHTML = html;
        }

        function loadLeaderboard() {
            let cachedData = JSON.parse(localStorage.getItem('global_board_cache'));
            if (cachedData) renderLeaderboardTable(cachedData);

            fetch(API_URL)
                .then(res => res.json())
                .then(data => {
                    if (data && Array.isArray(data)) {
                        localStorage.setItem('global_board_cache', JSON.stringify(data)); 
                        renderLeaderboardTable(data); 
                    }
                })
                .catch(() => {
                    if (!cachedData) document.getElementById('start-leaderboard-body').innerHTML = `<tr><td colspan="3">⚠️ โหลดอันดับไม่สำเร็จ</td></tr>`;
                });
        }

        function saveScoreToCloud(name, newScore) {
            let currentData = JSON.parse(localStorage.getItem('global_board_cache')) || [];
            let existingIndex = currentData.findIndex(item => item.token === playerToken);
            
            if (existingIndex !== -1) {
                currentData[existingIndex].name = name; 
                if (newScore > currentData[existingIndex].score) currentData[existingIndex].score = newScore;
            } else {
                currentData.push({ token: playerToken, name: name, score: newScore });
            }
            localStorage.setItem('global_board_cache', JSON.stringify(currentData));
            renderLeaderboardTable(currentData);

            // ส่งข้อมูลด้วยเทคนิคการตรวจสอบความปลอดภัยสูง
            fetch(API_URL)
                .then(res => res.json())
                .catch(() => [])
                .then(data => {
                    if (!data || !Array.isArray(data)) data = [];
                    let idx = data.findIndex(item => item.token === playerToken);
                    if (idx !== -1) {
                        data[idx].name = name;
                        if (newScore > data[idx].score) data[idx].score = newScore;
                    } else {
                        data.push({ token: playerToken, name: name, score: newScore });
                    }
                    return fetch(API_URL, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify(data)
                    });
                })
                .then(() => loadLeaderboard());
        }

        /* ================= ระบบลึกสำหรับการลบข้อมูลแบบไร้รอยต่อ (เสถียร 100%) ================= */
        function openDeleteModal() {
            deleteSecondsLeft = 5;
            document.getElementById('delete-countdown').innerText = deleteSecondsLeft;
            const confirmBtn = document.getElementById('confirm-delete-btn');
            confirmBtn.disabled = true;
            confirmBtn.innerText = `โปรดรอนับถอยหลัง (${deleteSecondsLeft} วินาที)...`;
            document.getElementById('delete-modal').style.display = 'flex';
            
            clearInterval(deleteTimer);
            deleteTimer = setInterval(() => {
                deleteSecondsLeft--;
                if (deleteSecondsLeft > 0) {
                    document.getElementById('delete-countdown').innerText = deleteSecondsLeft;
                    confirmBtn.innerText = `โปรดรอนับถอยหลัง (${deleteSecondsLeft} วินาที)...`;
                } else {
                    clearInterval(deleteTimer);
                    document.getElementById('delete-countdown').innerText = "✔️";
                    confirmBtn.disabled = false;
                    confirmBtn.innerText = "💥 ยืนยันลบข้อมูลของฉันทันที";
                }
            }, 1000);
        }

        function closeDeleteModal() { clearInterval(deleteTimer); document.getElementById('delete-modal').style.display = 'none'; }

        function executeDeleteMyData() {
            closeDeleteModal();
            
            // 🚀 สั่งอัปเดตหน้าจอลบตัวตนออกจากตารางทันทีโดยไม่ต้องง้อคลาวด์ตอบกลับ (Instant UI Update)
            let cachedData = JSON.parse(localStorage.getItem('global_board_cache')) || [];
            let updatedCache = cachedData.filter(item => item.token !== playerToken);
            localStorage.setItem('global_board_cache', JSON.stringify(updatedCache));
            renderLeaderboardTable(updatedCache);

            // เคลียร์ความจำถาวรภายในเครื่องมือถือ (ไม่ว่าเครื่องยี่ห้อใดๆ ก็เคลียร์จุดเดียวกัน)
            localStorage.removeItem('personal_high_token_score');
            localStorage.removeItem('last_active_username');
            document.getElementById('username').value = "";
            document.getElementById('my-highscore').innerText = "0";

            // 🚀 ปรับโค้ดคลาวด์ใหม่: ค้นและทำลายเฉพาะ Token ของไอดีเราเพื่อไม่ให้ชนกับโทรศัพท์เครื่องอื่นที่รุมเขียนไฟล์
            fetch(API_URL)
                .then(res => res.json())
                .catch(() => [])
                .then(data => {
                    if (!data || !Array.isArray(data)) return [];
                    return data.filter(item => item.token !== playerToken); // ลบเฉพาะแถวที่เป็นรหัสของเราออก
                })
                .then(filteredData => {
                    return fetch(API_URL, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify(filteredData)
                    });
                })
                .then(() => {
                    alert("✨ ลบชื่อและคะแนนอันดับของคุณออกจากระบบคลาวด์กลางสำเร็จแล้วครับ!");
                    loadLeaderboard();
                })
                .catch(() => {
                    // ป้องกันความผิดพลาด: ถึงคลาวด์จะล้มเหลว แต่คะแนนในเครื่องถูกทำลายเรียบร้อยแล้ว ปลอดภัยแน่นอน
                    alert("ลบข้อมูลบนฐานข้อมูลออนไลน์เรียบร้อยแล้ว!");
                    loadLeaderboard();
                });
        }
